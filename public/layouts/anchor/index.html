Anchor layout

Anchor layout allows the child containers to be anchored to a specific constraint.

 1package main
 2
 3import (
 4	"github.com/ebitenui/ebitenui"
 5	"github.com/ebitenui/ebitenui/image"
 6	"github.com/ebitenui/ebitenui/widget"
 7	"github.com/hajimehoshi/ebiten/v2"
 8	"golang.org/x/image/colornames"
 9)
10
11func main() {
12	ebiten.SetWindowSize(480, 320)
13	ebiten.SetWindowResizingMode(ebiten.WindowResizingModeEnabled)
14	if err := ebiten.RunGame(NewGame()); err != nil {
15		panic(err)
16	}
17}
18
19type Game struct {
20	ui *ebitenui.UI
21}
22
23func NewGame() *Game {
24	left := widget.NewContainer(
25		widget.ContainerOpts.BackgroundImage(
26			image.NewNineSliceColor(colornames.Lightskyblue),
27		),
28		widget.ContainerOpts.WidgetOpts(
29			widget.WidgetOpts.LayoutData(widget.AnchorLayoutData{
30				HorizontalPosition: widget.AnchorLayoutPositionStart,
31				StretchVertical:    true,
32			}),
33			widget.WidgetOpts.MinSize(50, 50),
34		),
35	)
36	right := widget.NewContainer(
37		widget.ContainerOpts.BackgroundImage(
38			image.NewNineSliceColor(colornames.Lightblue),
39		),
40		widget.ContainerOpts.WidgetOpts(
41			widget.WidgetOpts.LayoutData(widget.AnchorLayoutData{
42				HorizontalPosition: widget.AnchorLayoutPositionEnd,
43				StretchVertical:    true,
44			}),
45			widget.WidgetOpts.MinSize(50, 50),
46		),
47	)
48	up := widget.NewContainer(
49		widget.ContainerOpts.BackgroundImage(
50			image.NewNineSliceColor(colornames.Lightcyan),
51		),
52		widget.ContainerOpts.WidgetOpts(
53			widget.WidgetOpts.LayoutData(widget.AnchorLayoutData{
54				VerticalPosition:  widget.AnchorLayoutPositionStart,
55				StretchHorizontal: true,
56			}),
57			widget.WidgetOpts.MinSize(50, 50),
58		),
59	)
60	down := widget.NewContainer(
61		widget.ContainerOpts.BackgroundImage(
62			image.NewNineSliceColor(colornames.Lightslategrey),
63		),
64		widget.ContainerOpts.WidgetOpts(
65			widget.WidgetOpts.LayoutData(widget.AnchorLayoutData{
66				VerticalPosition:  widget.AnchorLayoutPositionEnd,
67				StretchHorizontal: true,
68			}),
69			widget.WidgetOpts.MinSize(50, 50),
70		),
71	)
72	root := widget.NewContainer(
73		widget.ContainerOpts.BackgroundImage(
74			image.NewNineSliceColor(colornames.Lightsteelblue),
75		),
76		widget.ContainerOpts.Layout(widget.NewAnchorLayout()),
77	)
78	root.AddChild(left)
79	root.AddChild(right)
80	root.AddChild(up)
81	root.AddChild(down)
82
83	return &Game{
84		ui: &ebitenui.UI{Container: root},
85	}
86}
87
88func (g *Game) Update() error {
89	g.ui.Update()
90	return nil
91}
92
93func (g *Game) Draw(screen *ebiten.Image) {
94	g.ui.Draw(screen)
95}
96
97func (g *Game) Layout(w, h int) (int, int) {
98	return w, h

Layout options

Padding

Layout allows you to specify padding for all child elements but not the itself.

Padding:
1root := widget.NewContainer(
2    widget.ContainerOpts.Layout(widget.NewAnchorLayout(
3        widget.AnchorLayoutOpts.Padding(widget.Insets{
4            Left: 50,
5        }),
6    )),
7)

1root := widget.NewContainer(

5        }),
